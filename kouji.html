// ============================================ // 工事スタッフ 業務日報 - Google Apps Script // ㈲ヤハタ電機商会 // ============================================ const SHEET_NAME = "工事日報"; // ★ 写真を保存するGoogle Driveのフォルダ ID を入れてください const DRIVE_FOLDER_ID = "17R6uVjwN6kei_VdLjNJX4LeA6TEINwc4"; const HEADERS = [ "日にち", "時間帯", "販売担当者", "責任者", "訪問先名(現場名)", "電話番号", "区分", "内容", "支払方法", "集金方法", "振込予定時期", "売上金額(千円)", "人数", "売上/人(千円)", "備考", "完了写真" ]; // -------------------------------------------- // POST リクエスト受信（テキスト＋写真対応） // -------------------------------------------- function doPost(e) { try { const contentType = e.parameter.contentType || ""; const data = JSON.parse(e.postData.contents); const spreadsheet = SpreadsheetApp.getActiveSpreadsheet(); let sheet = spreadsheet.getSheetByName(SHEET_NAME); // シートが存在しない場合は新規作成 if (!sheet) { sheet = spreadsheet.insertSheet(SHEET_NAME); sheet.appendRow(HEADERS); const headerRange = sheet.getRange(1, 1, 1, HEADERS.length); headerRange.setBackground("#1a3c5e"); headerRange.setFontColor("#ffffff"); headerRange.setFontWeight("bold"); headerRange.setHorizontalAlignment("center"); headerRange.setBorder(true, true, true, true, true, true); } // ---------- 写真の処理 ---------- let photoLinks = ""; if (data.photos && data.photos.length > 0) { const links = []; const folder = DriveApp.getFolderById(DRIVE_FOLDER_ID); // フォルダ名: 訪問先名_日付 const subFolderName = (data.date || "") + "_" + (data.visitName || "無名") + "_" + (data.detail || ""); let subFolder; // サブフォルダが存在するか確認 const existing = folder.getFoldersByName(subFolderName); if (existing.hasNext()) { subFolder = existing.next(); } else { subFolder = folder.createFolder(subFolderName); } data.photos.forEach((photo, idx) => { // base64からBlobに変換 const base64 = photo.split(",")[1]; // "data:image/..." のプレフィックスを除去 const mimeType = photo.split(";")[0].split(":")[1]; // "image/jpeg" など const fileName = (data.date || "") + "_" + (data.visitName || "無名") + "_" + (data.detail || "") + "_" + (idx + 1) + ".jpg"; const blob = Utilities.newBlob( Utilities.base64Decode(base64), mimeType, fileName ); const file = subFolder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); links.push(file.getUrl()); }); photoLinks = links.join("\n"); } // ---------- データを行にまとめる ---------- const row = [ data.date || "", data.time || "", data.sales || "", data.manager || "", data.visitName || "", data.phone || "", data.type || "", data.detail || "", data.payment || "", data.paymentMethod || "", data.paySchedule || "", data.amount || "0", data.people || "", data.perPerson || "", data.remarks || "", photoLinks ]; sheet.appendRow(row); // 追加した行のスタイル const lastRow = sheet.getLastRow(); const dataRange = sheet.getRange(lastRow, 1, 1, HEADERS.length); dataRange.setBorder(true, true, true, true, true, true); sheet.getRange(lastRow, 12, 1, 1).setHorizontalAlignment("right"); sheet.getRange(lastRow, 14, 1, 1).setHorizontalAlignment("right"); if (lastRow % 2 === 0) { dataRange.setBackground("#f9f9f9"); } return ContentService .createTextOutput(JSON.stringify({ result: "success" })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { console.log("エラー: " + err.getMessage()); return ContentService .createTextOutput(JSON.stringify({ result: "error", message: err.getMessage() })) .setMimeType(ContentService.MimeType.JSON); } } // -------------------------------------------- // GET リクエスト（テスト用） // -------------------------------------------- function doGet() { return ContentService .createTextOutput("工事スタッフ日報 GAS は正常に動作しています。") .setMimeType(ContentService.MimeType.TEXT); }
