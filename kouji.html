// ============================================ // 工事スタッフ 業務日報 - Google Apps Script // ㈲ヤハタ電機商会 // ============================================ // ★ シート名をここで設定してください const SHEET_NAME = "工事日報"; // ★ ヘッダー行の項目 const HEADERS = [ "日にち", "時間帯", "販売担当者", "責任者", "訪問先名(現場名)", "電話番号", "区分", "内容", "支払方法", "集金方法", "振込予定時期", "売上金額(千円)", "人数", "売上/人(千円)", "備考" ]; // -------------------------------------------- // POST リクエスト受信 // -------------------------------------------- function doPost(e) { try { const data = JSON.parse(e.postData.contents); const spreadsheet = SpreadsheetApp.getActiveSpreadsheet(); let sheet = spreadsheet.getSheetByName(SHEET_NAME); // シートが存在しない場合は新規作成 if (!sheet) { sheet = spreadsheet.insertSheet(SHEET_NAME); // ヘッダー行を追加 sheet.appendRow(HEADERS); // ヘッダー行のスタイル設定 const headerRange = sheet.getRange(1, 1, 1, HEADERS.length); headerRange.setBackground("#1a3c5e"); headerRange.setFontColor("#ffffff"); headerRange.setFontWeight("bold"); headerRange.setHorizontalAlignment("center"); headerRange.setBorder(true, true, true, true, true, true); } // データを配列にまとめる const row = [ data.date || "", data.time || "", data.sales || "", data.manager || "", data.visitName || "", data.phone || "", data.type || "", data.detail || "", data.payment || "", data.paymentMethod || "", data.paySchedule || "", data.amount || "0", data.people || "", data.perPerson || "", data.remarks || "" ]; // シートに行を追加 sheet.appendRow(row); // 追加した行のスタイル設定 const lastRow = sheet.getLastRow(); const dataRange = sheet.getRange(lastRow, 1, 1, HEADERS.length); dataRange.setBorder(true, true, true, true, true, true); // 売上金額と売上/人は右揃え sheet.getRange(lastRow, 12, 1, 1).setHorizontalAlignment("right"); // 売上金額 sheet.getRange(lastRow, 14, 1, 1).setHorizontalAlignment("right"); // 売上/人 // 売掛以外の場合は集金方法・振込予定時期をグレアウト if (data.payment !== "売掛") { sheet.getRange(lastRow, 10, 1, 2).setBackground("#f0f0f0"); } // 売掛かつ集金の場合は振込予定時期をグレアウト if (data.payment === "売掛" && data.paymentMethod === "集金") { sheet.getRange(lastRow, 11, 1, 1).setBackground("#f0f0f0"); } // 奇数行・偶数行で背景色を交互にする if (lastRow % 2 === 0) { dataRange.setBackground("#f9f9f9"); } return ContentService.createTextOutput(JSON.stringify({ result: "success" })) .setMimeType(ContentService.MimeType.JSON); } catch (e) { // エラーログを記録 console.log("エラー: " + e.getMessage()); return ContentService.createTextOutput(JSON.stringify({ result: "error", message: e.getMessage() })) .setMimeType(ContentService.MimeType.JSON); } } // -------------------------------------------- // GET リクエスト（テスト用） // ブラウザでURLにアクセスして動作確認できる // -------------------------------------------- function doGet() { return ContentService.createTextOutput("工事スタッフ日報 GAS は正常に動作しています。") .setMimeType(ContentService.MimeType.TEXT); } // -------------------------------------------- // 月次レポート（手動実行用） // シートの末尾に当月の合計を追加する // -------------------------------------------- function addMonthlyTotal() { const spreadsheet = SpreadsheetApp.getActiveSpreadsheet(); const sheet = spreadsheet.getSheetByName(SHEET_NAME); if (!sheet) { SpreadsheetApp.getUi().alert("シートが見つかりません"); return; } const lastRow = sheet.getLastRow(); const today = new Date(); const year = today.getFullYear(); const month = today.getMonth() + 1; const monthStr = year + "-" + String(month).padStart(2, "0"); let totalAmount = 0; let totalVisits = 0; // 2行目以降（ヘッダー除く）を走査 for (let i = 2; i <= lastRow; i++) { const dateVal = sheet.getRange(i, 1).getString(); if (dateVal.startsWith(monthStr)) { totalAmount += Number(sheet.getRange(i, 12).getString()) || 0; totalVisits++; } } // 合計行を追加 const totalRow = lastRow + 2; sheet.getRange(totalRow, 1).setValue("【" + monthStr + " 月次合計】"); sheet.getRange(totalRow, 7).setValue("訪問件数: " + totalVisits + "件"); sheet.getRange(totalRow, 12).setValue(totalAmount); // スタイル設定 const totalRange = sheet.getRange(totalRow, 1, 1, HEADERS.length); totalRange.setBackground("#fff3e0"); totalRange.setFontWeight("bold"); totalRange.setBorder(true, true, true, true, true, true); SpreadsheetApp.getUi().alert( monthStr + " の月次合計を追加しました。\n" + "訪問件数: " + totalVisits + "件\n" + "売上合計: " + totalAmount + " 千円" ); } // -------------------------------------------- // シート幅の自動調整（手動実行用） // -------------------------------------------- function autoResize() { const spreadsheet = SpreadsheetApp.getActiveSpreadsheet(); const sheet = spreadsheet.getSheetByName(SHEET_NAME); if (!sheet) { SpreadsheetApp.getUi().alert("シートが見つかりません"); return; } sheet.autoResizeColumns(1, HEADERS.length); SpreadsheetApp.getUi().alert("列幅を自動調整しました。"); }
